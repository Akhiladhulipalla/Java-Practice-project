package Arrays;

public class Arrays {
	public static void main(String[] args) {
		int[] values = {20,30,40,50,60};
		
		System.out.println("Accessing Elements from Array :");
		System.out.println("First Element :"+values[0]);
		System.out.println("Second Element :;"+values[1]);
		System.out.println("Third Element :"+values[2]);
		System.out.println("Fourth Element :"+values[3]);
		System.out.println("Fifth Element :"+values[4]);
	}

}


Output:
======

Accessing Elements from Array :
First Element :20
Second Element :;30
Third Element :40
Fourth Element :50
Fifth Element :60
